Histogram2d(
    x = x,
    y = y,
    histnorm = "probability",
    autobinx = false,
    xbins =
        Xbins(
            start = -3.,
            ``end`` = 3.,
            size = 0.1
        ),
    autobiny = false,
    ybins =
        Ybins(
            start = -2.5,
            ``end`` = 4.,
            size = 0.1
        ),
    colorscale =
        [
            [box 0; box "rgb(12,51,131)"]
            [0.25; "rgb(10,136,186)"]
            [0.5; "rgb(242,211,56)"]
            [0.75; "rgb(242,143,56)"]
            [1; "rgb(217,30,30)"]
        ]
)
|> Chart.Plot
|> Chart.WithWidth 700
|> Chart.WithHeight 500

let trace1 =
    Scatter(
        x = x0,
        y = y0,
        mode = "markers",
        marker =
            Marker(
                symbol = "circle",
                opacity = 0.7
            )
    )

let trace2 =
    Scatter(
        x = x1,
        y = y1,
        mode = "markers",
        marker =
            Marker(
                symbol = "square",
                opacity = 0.7
            )
    )

let trace3 =
    Histogram2d(
        x = x',
        y = y'
    )

let layout = Layout(showlegend = false)

[trace1 :> Trace; trace2 :> _; trace3 :> _]
|> Chart.Plot
|> Chart.WithWidth 700
|> Chart.WithHeight 500
|> Chart.WithLayout layout
